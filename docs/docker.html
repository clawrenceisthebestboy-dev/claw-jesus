Quick Start

Docker Compose
Docker Run
GPU (NVIDIA)
# Download the compose file
curl -O https://clawrenceisthebestboy-dev.github.io/saferclaw/docker/docker-compose.yml

# Set your license key
export OPENCLAW_LICENSE=your-license-key

# Start everything
docker compose up -d
docker run -d \
  -p 3000:3000 \
  -p 11434:11434 \
  -v openclaw_data:/root/.openclaw \
  -v ollama_data:/root/.ollama \
  -e OPENCLAW_LICENSE=your-license-key \
  -e OLLAMA_MODEL=llama3.2 \
  saferclaw/saferclaw:latest
docker run -d \
  --gpus all \
  -p 3000:3000 \
  -p 11434:11434 \
  -v openclaw_data:/root/.openclaw \
  -v ollama_data:/root/.ollama \
  -e OPENCLAW_LICENSE=your-license-key \
  -e OLLAMA_MODEL=llama3.2 \
  saferclaw/saferclaw:latest
